  \centering{%
    {\primaryfontmedium{28pt}#1 #2. #3}%
  } \\[5pt]%
}

\newcommand{\headersubtitle}[1]{%

    \color{headings}
    \primaryfontlight{11pt}#1
  }
  \par\headerskip
}
